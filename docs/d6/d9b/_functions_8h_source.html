218inline float approximateSine(float angle)
219{
220 float x,j;
221
222 if (angle < PIo2){
223 x = angle * TWOoPI - 0.5f;
224 j = -(x * x) + 0.75f + x;
225 }
226 else if(angle < PI) {
227 angle = PI - angle;
228 x = angle * TWOoPI - 0.5f;
229 j = -(x * x) + 0.75f + x;
230 }
231 else if (angle < PI3o2) {
232 angle -= PI;
233 x = angle * TWOoPI - 0.5f;
234 j = x * x - 0.75f - x;
235 }
236 else {
237 angle = TWOPI - angle;
238 x = angle * TWOoPI - 0.5f;
239 j = x * x - 0.75f - x;
240 }
241
242 return j;
243}

265inline float approximateTanh(const float x)
266{
267 float sign = getSign(x);
268 float input = (sign > 0.f) ? x : -x;
269
270 if (input > 8.f) return 1.f * (float)sign;
271
272 float readPointer = mapValue(input, 0.f, 8.f, 0.f, 4095.f);
273 int readPointerLo = (int)readPointer;
274 int readPointerHi = readPointerLo + 1;
275 float frac = readPointer - (float)readPointerLo;
276
277 float lowValue = TANH_WAVETABLE_POSITIVEONLY_4096[readPointerLo];
278
279 if (frac == 0.f) return lowValue * (float)sign;
280
281 float highValue = (readPointerHi < 4096) ? TANH_WAVETABLE_POSITIVEONLY_4096[readPointerHi] : 1.f;
282
283 return (lowValue + frac * (highValue - lowValue)) * sign;
284}

552inline float generateGaussian(const float mean, const float stddev)
553{
554 static bool haveSpare = false;
555 static float rand1, rand2;
556
557 if (haveSpare)
558 {
559 haveSpare = false;
560 return mean + stddev * sqrtf_neon(rand1) * sinf_neon(rand2);
561 }
562
563 haveSpare = true;
564
565 rand1 = rand() * RAND_MAX_INVERSED;
566 if (rand1 < 1e-100) rand1 = 1e-100; // Avoid log(0)
567
568 rand1 = -2.f * logf_neon(rand1);
569 rand2 = (rand() * RAND_MAX_INVERSED) * 2.0f * PI;
570
571 return mean + stddev * sqrtf_neon(rand1) * cosf_neon(rand2);
572}
