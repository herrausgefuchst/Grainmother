74class SimpleDelayStereo
75{
76public:
84 void setup(const float& delaySamples_, const int& maxDelaySamples_, const float& sampleRate_)
85 {
86 // adjust length of buffer to the next fitting power of 2
87 bufferLength = (unsigned int)powf_neon(2.f, 1.f + (ceilf_neon(logf_neon(maxDelaySamples_) / log_2)));
88
89 // used for wrapping the pointers
90 bufferWrap = bufferLength - 1;
91
92 // initialize the buffer
93 buffer.reset(new float32x2_t[bufferLength]);
94
95 // set all values in buffer to 0.f
96 std::fill(buffer.get(), buffer.get()+bufferLength, vdup_n_f32(0.f));
97
98 // setup readPointer
99 setDelay(delaySamples_);
100 }
101
108 float32x2_t processAudioSamples(const float32x2_t& input_)
109 {
110 // write buffer
111 buffer[writePointer] = input_;
112
113 // increase pointers
114 writePointer = (writePointer + 1) & bufferWrap;
115 readPointerLo = (readPointerLo + 1) & bufferWrap;
116 readPointerHi = (readPointerHi + 1) & bufferWrap;
117
118 // read buffer
119 float32x2_t returnValue = buffer[readPointerLo];
120 if (interpolationNeeded) {
121 float32x2_t diff = vsub_f32(buffer[readPointerHi], returnValue);
122 returnValue = vmla_n_f32(returnValue, diff, frac);
123 }
124 return returnValue;
125 }
126
132 void setDelay(const float& delaySamples_)
133 {
134 // set and bound delay, delay in samples should not ecxeed bufferlength
135 delaySamples = delaySamples_ + 1.f;
136 boundValue(delaySamples, 0.f, (float)bufferWrap);
137
138 // interpolation stuff
139 readPointerLo = writePointer - (int)delaySamples;
140 readPointerHi = readPointerLo - 1;
141 if (readPointerLo < 0) readPointerLo += bufferLength;
142 if (readPointerHi < 0) readPointerHi += bufferLength;
143 frac = delaySamples - (int)delaySamples;
144 interpolationNeeded = frac == 0.f ? false : true;
145 }
146
147 float getDelay() const { return delaySamples; }
148
149private:
150 std::unique_ptr<float32x2_t[]> buffer = nullptr;
151 unsigned int bufferLength = 1024;
152 unsigned int bufferWrap = 1023;
153 unsigned int writePointer = 0;
154 int readPointerLo = 0;
155 int readPointerHi = 0;
156 float frac = 0.f;
157 bool interpolationNeeded = false;
158 float delaySamples = 0;
159};

228class ButterworthHighcutStereo
229{
230public:
232 void setup(const float& cutoffFreq_, const float& sampleRate_)
233 {
234 // calc inversed nyquist frequency
235 nyquist_inv = 1.f / (sampleRate_ * 0.5f);
236
237 // setup cutoff frequency
238 setCutoffFrequency(cutoffFreq_);
239 }
240
242 void setCutoffFrequency(const float& cutoffFreq_)
243 {
244 // set and bound value
245 cutoffFrequency = cutoffFreq_;
246 boundValue(cutoffFrequency, 100.f, 20000.f);
247
248 // calculate filter coefficients
249 calculateCoefficients();
250
251 // set on/off flag
252 enabled = cutoffFrequency < 20000 ? true : false;
253 }
254
256 void processAudioSamples(float32x2_t& x_)
257 {
258 // y(n) = b0 * x(n) + b1 * x(n-1) + b2 * x(n-2) - a1 * y(n-1) - a2 * y(n-2)
259 float32x2_t y = vadd_f32(vmul_n_f32(x_, b0), vmul_n_f32(x1, b1));
260 y = vmla_n_f32(y, x2, b2);
261 y = vmls_n_f32(y, y1, a1);
262 y = vmls_n_f32(y, y2, a2);
263
264 // delay the state variables
265 x2 = x1;
266 x1 = x_;
267 y2 = y1;
268 y1 = y;
269
270 // overwrite the input
271 x_ = y;
272 }
273
275 const float& getCutoffFrequency() const { return cutoffFrequency; }
276
277private:
279 void calculateCoefficients()
280 {
281 float normal_cutoff = cutoffFrequency * nyquist_inv;
282 float omega_c = PI * normal_cutoff;
283 float tan_half_wc = tanf_neon(omega_c * 0.5f);
284
285 float b0_coef = tan_half_wc * tan_half_wc;
286
287 // sets resonance, lower cutoff frequencys = more resonance
288 float Q = 1.0f + (40.f / (cutoffFrequency + 1.0f));
289
290 float sqrt2tanhalfwc = sqrt_2 * tan_half_wc / Q;
291 float denom_inv = 1.f / (1.0f + sqrt2tanhalfwc + b0_coef);
292
293 b0 = b0_coef * denom_inv;
294 b1 = 2.0f * b0;
295 b2 = b0;
296 a1 = 2.0f * (b0_coef - 1.0f) * denom_inv;
297 a2 = (1.0f - sqrt2tanhalfwc + b0_coef) * denom_inv;
298 }
299
300private:
301 float cutoffFrequency;
302 float nyquist_inv;
303
305 float32_t b0, b1, b2, a1, a2;
306
308 float32x2_t x1 = vdup_n_f32(0.f), x2 = vdup_n_f32(0.f);
309 float32x2_t y1 = vdup_n_f32(0.f), y2 = vdup_n_f32(0.f);
310
311public:
312 bool enabled = false;
313};

416class ParametricEQStereo
417{
418public:
420 void setup(const float& centerFreq_, const float& gain_, const float& bandwidth_, const float& sampleRate_)
421 {
422 // inversed sample rate
423 fs_inv = 1.f / sampleRate_;
424
425 // bandwidth
426 bandwidth = bandwidth_;
427 bandwidth2 = 2.f * bandwidth_;
428
429 // gain
430 gain = gain_;
431 A = powf_neon(10.0f, gain * 0.025f);
432 A_o1 = 1.f / A;
433
434 // center frequency
435 setCenterFrequency(centerFreq_);
436 }
437
439 void setCenterFrequency(const float& centerFreq_)
440 {
441 // set and bound value
442 centerFreq = centerFreq_;
443 boundValue(centerFreq, 20.f, 20000.f);
444
445 // calculate helper variables
446 omega0 = TWOPI * centerFreq * fs_inv;
447 cosOmega0 = cosf_neon(omega0);
448 sinOmega0 = sinf_neon(omega0);
449 alpha = sinOmega0 * sinhf_neon(log_2 / bandwidth2 * omega0 / sinOmega0);
450
451 // recalculate filter coefficients
452 calculateCoefficients();
453 }
454
456 void setGain(const float& gain_)
457 {
458 // set and bound value
459 gain = gain_;
460 boundValue(gain, -12.f, 12.f);

464 A_o1 = 1.f / A;
465
466 // recalculate filter coefficients
467 calculateCoefficients();
468
469 // set on/off flag
470 enabled = gain != 0.f ? true : false;
471 }
472
474 void setBandwidth(const float& bandwidth_)
475 {
476 // set
477 bandwidth = bandwidth_;
478
479 // calculate helper variables
480 bandwidth2 = 2.0f * bandwidth;
481 alpha = sinOmega0 * sinhf_neon(log_2 / bandwidth2 * omega0 / sinOmega0);
482
483 // recalculate filter coefficients
484 calculateCoefficients();
485 }
486
488 void processAudioSamples(float32x2_t& x_)
489 {
490 // y(n) = b0 * x(n) + b1 * x(n-1) + b2 * x(n-2) - a1 * y(n-1) - a2 * y(n-2)

499 y2 = y1;

513 {
514 float denominator_inv = 1.f / (1.0f + alpha * A_o1);
