214class HighPassFilter
215{
216public:
217 void setup(float cutoffFreq_, float sampleRate_)
218 {
219 sampleRate = sampleRate_;
220 setCutoffFrequency(cutoffFreq_);
221 reset();
222 }
223
224 void setCutoffFrequency(float cutoffFreq)
225 {
226 float omega = 2.0f * PI * cutoffFreq / sampleRate;
227 float alpha = sinf(omega) / (2.0f * sqrtf(2.0f)); // Q = sqrt(2)/2 for 12dB/oct Butterworth
228
229 float cos_omega = cosf(omega);
230
231 // Filter coefficients (biquad)
232 float b0 = (1.0f + cos_omega) / 2.0f;
233 float b1 = -(1.0f + cos_omega);
234 float b2 = (1.0f + cos_omega) / 2.0f;
235 float a0 = 1.0f + alpha;
236 float a1 = -2.0f * cos_omega;
237 float a2 = 1.0f - alpha;
238
239 // Normalize coefficients by a0 and initialize NEON vectors
240 b0v = vdup_n_f32(b0 / a0);
241 b1v = vdup_n_f32(b1 / a0);
242 b2v = vdup_n_f32(b2 / a0);
243 a1v = vdup_n_f32(a1 / a0);
244 a2v = vdup_n_f32(a2 / a0);
245 }
246
247 StereoFloat process(StereoFloat input_)
248 {
249 float32x2_t inputv = { input_[0], input_[1] };
250
251 // Calculate output = b0*input + b1*x1 + b2*x2 - a1*y1 - a2*y2 (NEON parallel computation)
252 float32x2_t output = vmla_f32(vmla_f32(vmul_f32(b0v, inputv), b1v, x1), b2v, x2);
253 output = vmls_f32(vmls_f32(output, a1v, y1), a2v, y2);
254
255 // Shift history of samples for both channels
256 x2 = x1;
257 x1 = inputv;
258 y2 = y1;
259 y1 = output;
260
261 return { vget_lane_f32(output, 0), vget_lane_f32(output, 1) };
262 }
263
264 // Reset the filter history (for initializing or resetting the filter)
265 void reset()
266 {
267 x1 = vdup_n_f32(0.0f);
268 x2 = vdup_n_f32(0.0f);
269 y1 = vdup_n_f32(0.0f);
270 y2 = vdup_n_f32(0.0f);
271 }
272
273private:
274 // Filter coefficients as NEON vectors
275 float32x2_t b0v, b1v, b2v, a1v, a2v;
276
277 // Previous input and output samples (for history) as NEON vectors
278 float32x2_t x1, x2, y1, y2;
279
280 // Sample rate as a member variable
281 float sampleRate;
282};

504class Delay
505{
506public:
512 Delay()
513 {
514 std::fill(buffer.begin(), buffer.end(), vdup_n_f32(0.f));
515 }
516
524 void setup(const float sampleRate_)
525 {
526 sampleRate = sampleRate_;
527 delayMs.setup(100.f, sampleRate, 8);
528 }
529
540 float32x2_t processAudioSamples(float32x2_t input_, const uint sampleIndex_)
541 {
542 if ((sampleIndex_ & 7) == 0)
543 {
544 if (!delayMs.rampFinished) delayMs.processRamp();
545 setDelayTimeInMs(delayMs());
546 }
547
548 float32x2_t output = buffer.at(readPointerLo);
549
550 // Linear interpolation
551 if (interpolationNeeded)
552 {
553 float32x2_t interpolated = vmul_n_f32(vsub_f32(buffer.at(readPointerHi), output), frac);
554 output = vadd_f32(output, interpolated);
555 }
556
557 buffer.at(writePointer) = vmla_n_f32(vrev64_f32(input_), output, feedback);
558
559 if (++writePointer >= bufferLength) writePointer = 0;
560 if (++readPointerLo >= bufferLength) readPointerLo = 0;
561 if (++readPointerHi >= bufferLength) readPointerHi = 0;
562
563 return output;
564 }
565
571 void setFeedback(const float32_t feedback_) { feedback = feedback_; }
572
580 void setDelayTimeRampInMs(const float delayMs_)
581 {
582 delayMs.setRampTo(delayMs_, 0.1f);
583 }
584
592 void setDelayTimeInSamples(const uint delaySamples_)
593 {
594 if (delaySamples_ >= bufferLength)
595 engine_rt_error("delay exceeds buffer length of delay object", __FILE__, __LINE__, true);
596
597 readPointerLo = writePointer - delaySamples_;
598 if (readPointerLo < 0) readPointerLo += bufferLength;
599
600 interpolationNeeded = false;
601 }
602
611 void setDelayTimeInMs(const float delayMs_)
612 {
613 float delaySamples = delayMs_ * 0.001f * sampleRate;
614
615 readPointerLo = writePointer - floorf_neon(delaySamples);
616 readPointerHi = readPointerLo + 1;
617
618 if (readPointerLo < 0) readPointerLo += bufferLength;
619 if (readPointerHi < 0) readPointerHi += bufferLength;
620
621 frac = delaySamples - floorf_neon(delaySamples);
622 interpolationNeeded = (frac == 0.f) ? false : true;
623 }
624
625private:
626 float sampleRate = 44100.f;
627
628 LinearRamp delayMs;
629
630 static const uint bufferLength = 65536;
631 std::array<float32x2_t, bufferLength> buffer;
632
633 uint writePointer = 0;
634
635 int readPointerLo, readPointerHi;
636 float32_t frac;
637 bool interpolationNeeded = false;

639 float32_t feedback;
